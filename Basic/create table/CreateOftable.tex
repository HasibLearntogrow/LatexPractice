%---------------------------------------------------------merge colom---------------------------------------------------------------------------------------------

\section{Create table with merge colum:}
\begin{table}[h]
	\centering
	\begin{tabular}{|c|c|c|}
		\hline
		Name & Address & Phone number \\ 
		\hline
		\multicolumn{2}{|c|}{John Doe} & 555-1234 \\
		 \hline
		Jane Smith & 456 Oak Ave. & 555-5678 \\
		 \hline
	\end{tabular}
	\caption{Contact Information}
\end{table}
